```sql
-- meta_daily_insights indexes for env-scoped historical-window queries
--
-- aggregate_meta_daily_insights (latest migration: 20260522120000_...) filters
-- by (environment, date_start range) and LEFT JOINs to env-scoped metadata by
-- (meta_account_id, campaign_id), (meta_account_id, ad_set_id),
-- (meta_account_id, ad_id), and (brand_id). Without a composite index covering
-- (environment, date_start), full-year queries hit statement_timeout.
--
-- Uses CONCURRENTLY to avoid blocking writes during creation. Cannot run inside
-- a transaction, so this migration uses no explicit BEGIN/COMMIT.

-- Primary lookup index: env + date range
create index concurrently if not exists meta_daily_insights_env_date_idx
  on public.meta_daily_insights (environment, date_start);

-- Covering index for the env-scoped LEFT JOINs. Order columns by selectivity:
-- environment (2 values), date_start (range), then JOIN keys.
create index concurrently if not exists meta_daily_insights_env_date_join_idx
  on public.meta_daily_insights (environment, date_start, brand_id, meta_account_id, campaign_id, ad_set_id, ad_id);
```

export async function fetchCustomerJourneyLedgerData(range: DateRange) {
  // 1. Fetch all visitors active in the window (NEW — was appointment-keyed)
  const visitors = await supabase
    .from("website_visitors")
    .select("visitor_id, first_seen_at, last_seen_at, country, region, city")
    .gte("last_seen_at", range.start)
    .lte("last_seen_at", range.end)
    .limit(MAX_LEDGER_VISITORS);

  // 2. Fetch appointments in window (KEEP — for booking-side rows)
  const appointments = await supabase
    .from("appointment_events")
    .select("acuity_appointment_id, visitor_id, visit_date_time, ...")
    .gte("visit_date_time", range.start)
    .lte("visit_date_time", range.end);

  // 3. UNION-style: for each visitor, find their appointments (if any).
  //    For each orphan appointment (no visitor row), include as appointment-only row.
  const visitorIds = new Set(visitors.data?.map((v) => v.visitor_id) ?? []);
  const apptByVisitor = new Map<string, any[]>();
  const orphanAppts: any[] = [];
  for (const appt of appointments.data ?? []) {
    if (appt.visitor_id && visitorIds.has(appt.visitor_id)) {
      const list = apptByVisitor.get(appt.visitor_id) ?? [];
      list.push(appt);
      apptByVisitor.set(appt.visitor_id, list);
    } else {
      orphanAppts.push(appt);
    }
  }

  // 4. Fetch related events/sessions/conversions for visitors (in batches as before)
  // ... (keep existing batched fetches but key by visitor first)

  // 5. Compose ledger rows: one per visitor (with their appointments inlined),
  //    one per orphan appointment.
  return {
    rows: [
      ...visitors.data!.map((v) => composeVisitorRow(v, apptByVisitor.get(v.visitor_id) ?? [], ...)),
      ...orphanAppts.map((a) => composeAppointmentOrphanRow(a, ...)),
    ],
    // ... other return fields unchanged
  };
}
